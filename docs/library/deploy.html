    # 示例

    ###
# @Author: shenlan
# @LastEditors: shenlan
# @Description: 打包上线脚本
# @启动方式
#
# sh deploy.sh build        --------正式上线
# sh deploy.sh build:alpha   --------线上测试
# sh deploy.sh build:release  --------预上线(alpha测试通过后，先用release测试功能，通过后再正式上线)--(预留功能当前未使用)
#
###
#!/bin/sh

# 设置变量

# 脚本命令
BUILD_COMMAND="npm run $1"

# 打包后输出文件夹
DIST_FOLDER="dist"

# 本地文件夹名称
LOCAL_FOLDER_NAME="xxxxx"

# 本地文件夹目录
LOCAL_DIRECTORY="xxxxxx/$LOCAL_FOLDER_NAME/dist"

# 服务器文件夹名称
SERVER_FOLDER_NAME="xxxxx"

# 远程服务器
REMOTE_SERVER="root@xx.xxx.xx.xxx"

# 远程服务器文件夹存放位置
REMOTE_DIRECTORY="xxxxxxx"

# 远程目录文件夹
REMOTE_DIRECTORY_FOLDER="$REMOTE_DIRECTORY/$SERVER_FOLDER_NAME"

# 日志记录函数
log() {
    echo '\033[32m\003'
    echo "$(date +%Y-%m-%d\ %H:%M:%S) >>> $1"
}

# 错误处理函数
error() {
    log "ERROR: $1"
    exit 1
}

log "***************开始执行脚本***************"

# 删除旧文件
log "删除 $DIST_FOLDER 文件夹和压缩包"
rm -rf $DIST_FOLDER
rm -f $DIST_FOLDER-*.zip

# 运行构建命令
log "运行命令: $BUILD_COMMAND"
$BUILD_COMMAND || error "Build failed"

log "$BUILD_COMMAND 执行完成"

# 线上生产环境
if [ $1 = "build" ]; then
    log "当前是线上生产环境 $1---------------"
    log '开始压缩文件夹 dist'
    echo ''

    # 创建压缩文件
    zip -r $DIST_FOLDER-$(date +%Y-%m-%d_%H-%M-%S).zip $DIST_FOLDER || error "压缩文件夹失败"

    log '压缩完成'
    log "当前是线上生产环境 $1---------------"

#线上开发测试环境 (开发自己用)
elif [ $1 = "build:alpha" ]; then
    log "当前是线上测试环境 $1 ---------------"

    # # 复制服务器文件到本地
    # echo ''
    # if scp "$REMOTE_SERVER:$REMOTE_DIRECTORY_TAISHAN/Readme.txt" "$LOCAL_DIRECTORY"; then
    #     echo "复制成功"
    # else
    #     echo "复制失败"
    #     exit 1
    # fi

    # 移除服务器文件夹并新建文件夹
    log "移除并新建文件夹 $SERVER_FOLDER_NAME"
    if ssh "$REMOTE_SERVER" "cd $REMOTE_DIRECTORY && rm -rf $SERVER_FOLDER_NAME && mkdir $SERVER_FOLDER_NAME"; then
        log "$SERVER_FOLDER_NAME 文件夹移除并新建成功"
    else
        log "错误: 文件夹移除或新建失败"
        exit 1
    fi

    # 上传文件
    log "开始上传文件"
    if scp -r "$LOCAL_DIRECTORY"/* "$REMOTE_SERVER:$REMOTE_DIRECTORY_FOLDER"; then
        log "文件成功被上传到 $REMOTE_DIRECTORY_FOLDER"
    else
        log "错误: 文件上传失败 $REMOTE_DIRECTORY_FOLDER"
        exit 1
    fi
    log "上传文件结束"
    log "当前是线上测试环境 $1---------------"

fi
#ifend

log "*************** 脚本执行完成 ***************"
echo ''
echo ''
echo ''
