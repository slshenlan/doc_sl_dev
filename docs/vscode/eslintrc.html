      0,
      {
        properties: 'always' // 属性名称强制执行驼峰命名法样式

      2,
      {

        after: true // 在逗号后需要一个或多个空格

        before: true,

    'no-dupe-args': 2,

    'no-dupe-class-members': 2,

    'no-duplicate-case': 2,

    'no-empty-pattern': 2,

    'no-extend-native': 2,

    'no-extra-bind': 2,

    'no-extra-boolean-cast': 2,

    'no-mixed-spaces-and-tabs': 2,

    'no-obj-calls': 2,

    'no-octal': 2,

    'no-octal-escape': 2,

    'no-path-concat': 2,

    'no-proto': 2,

    'no-redeclare': 2,

    'no-regex-spaces': 2,

    'no-self-assign': 2,

    'no-self-compare': 2,

    'no-sequences': 2,

    'no-shadow-restricted-names': 2,
    // 功能标识符与其应用程序之间不允许有间距
    'no-spaced-func': 2,
