    # 获取 url 中"?"符后的字串

    slGetQuery

    import { slGetQuery } from 'shenlan-util';

// 不传参数
slGetQuery(); // -> 当前url上携带的参数

// 传参数
slGetQuery('https://ifengniao.com?a=1&b=2&c=3'); // -> {a:1,b:2,c:3}
