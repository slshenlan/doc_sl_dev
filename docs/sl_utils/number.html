    # 随机整数生成

    slRandomNumber

    import { slRandomNumber } from 'shenlan-util';

/**
 * @description: 随机整数生成
 * @param {Number} n 开始的数字
 * @param {Number} m 结束的数字
 * @param {Number} type 类型
 * type  1 生成n-m，包含n但不包含m的整数
 *       2 生成n-m，不包含n但包含m的整数
 *       3 生成n-m，不包含n和m的整数
 *       4 生成n-m，包含n和m的随机数
 * @return {Number}
 */
slRandomNumber(1, 9);
